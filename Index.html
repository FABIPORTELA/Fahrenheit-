def fahrenheit_para_celsius(fahrenheit): return (fahrenheit - 32) * 5.0/9.0 def celsius_para_fahrenheit(celsius): return (celsius * 9.0/5.0) + 32 def main(): print("Escolha a conversão:") print("1: Fahrenheit para Celsius") print("2: Celsius para Fahrenheit") escolha = input("Digite 1 ou 2: ") if escolha == '1': fahrenheit = float(input("Digite a temperatura em Fahrenheit: ")) celsius = fahrenheit_para_celsius(fahrenheit) print(f"{fahrenheit}°F é igual a {celsius:.2f}°C") elif escolha == '2': celsius = float(input("Digite a temperatura em Celsius: ")) fahrenheit = celsius_para_fahrenheit(celsius) print(f"{celsius}°C é igual a {fahrenheit:.2f}°F") else: print("Escolha inválida. Por favor, escolha 1 ou 2.") if __name__ == "__main__": main()
